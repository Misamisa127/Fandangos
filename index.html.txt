from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
from selenium.webdriver.chrome.service import Service
from webdriver_manager.chrome import ChromeDriverManager
import time

def login_sala_do_futuro(driver, ra, senha):
    driver.get("https://saladofuturo.educacao.sp.gov.br")
    time.sleep(2)
    
    campo_ra = driver.find_element(By.NAME, "ra")
    campo_senha = driver.find_element(By.NAME, "senha")
    botao_login = driver.find_element(By.XPATH, "//button[@type='submit']")
    
    campo_ra.send_keys(ra)
    campo_senha.send_keys(senha)
    botao_login.click()
    time.sleep(5)

def responder_atividade(driver):
    atividades = driver.find_elements(By.CLASS_NAME, "atividade-pendente")
    for atividade in atividades:
        atividade.click()
        time.sleep(3)
        
        botao_ver_resposta = driver.find_element(By.XPATH, "//button[text()='Ver resposta correta']")
        botao_ver_resposta.click()
        time.sleep(2)
        
        resposta_correta = driver.find_element(By.CLASS_NAME, "resposta-correta").text
        print(f"Resposta correta: {resposta_correta}")  # Exibe a resposta correta no terminal
        
        opcoes = driver.find_elements(By.CLASS_NAME, "opcao-resposta")
        
        for opcao in opcoes:
            if opcao.text.strip() == resposta_correta.strip():
                opcao.click()
                break
        
        botao_finalizar = driver.find_element(By.XPATH, "//button[text()='Finalizar']")
        botao_finalizar.click()
        time.sleep(2)

def main():
    ra = "seu_ra_aqui"
    senha = "sua_senha_aqui"
    
    driver = webdriver.Chrome(service=Service(ChromeDriverManager().install()))
    try:
        login_sala_do_futuro(driver, ra, senha)
        responder_atividade(driver)
    finally:
        driver.quit()

if __name__ == "__main__":
    main()
